import pandas as pd
from sklearn.preprocessing import MultiLabelBinarizer
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report

import matplotlib.pyplot as plt

# 3.1 Load your transactions data
data_dir = './8451_The_Complete_Journey_2_Sample-2-1/8451_The_Complete_Journey_2_Sample-2'
tx = pd.read_csv(f'{data_dir}/400_transactions.csv')
tx.columns = tx.columns.str.strip()
prod = pd.read_csv(f'{data_dir}/400_products.csv')
prod.columns = prod.columns.str.strip()

# 3.2 Merge so each transaction has a COMMODITY label
tx = tx.merge(prod[['PRODUCT_NUM', 'COMMODITY']], on='PRODUCT_NUM', how='left')

# 3.3 Build baskets of commodities
baskets = tx.groupby('BASKET_NUM')['COMMODITY'].apply(list)

# 3.4 One-hot encode commodity baskets (dense  85k42  ~27 MB)
mlb = MultiLabelBinarizer()
X = mlb.fit_transform(baskets)

# 3.5 Pick the most-common commodity as the target
target_comm = tx['COMMODITY'].value_counts().idxmax()
print(f"Predicting purchase of COMMODITY {target_comm}")

# 3.6 Build labels & drop empty baskets
y = baskets.apply(lambda items: int(target_comm in items)).values
nonempty = X.sum(axis=1) > 0
X, y = X[nonempty], y[nonempty]

# 3.7 Split into train/test
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42
)

# 3.8 Train a lean Random Forest
rf = RandomForestClassifier(
    n_estimators=30,
    max_features='sqrt',
    n_jobs=-1,
    random_state=42
)
rf.fit(X_train, y_train)
y_pred = rf.predict(X_test)

# 3.9 Evaluate
print("Basket Analysis Classification Report:")
print(classification_report(y_test, y_pred))

# 3.10 Plot top-10 commodity co-purchase predictors (excluding the target)
importances = pd.Series(rf.feature_importances_, index=mlb.classes_)
importances = importances.drop(target_comm).nlargest(10)

plt.figure(figsize=(8, 5))
importances.sort_values().plot.barh()
plt.title(f"Top 10 Co-purchase Commodities for {target_comm}")

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import classification_report, roc_auc_score, roc_curve
import matplotlib.pyplot as plt

# 1) Load & clean data
data_dir = './8451_The_Complete_Journey_2_Sample-2-1/8451_The_Complete_Journey_2_Sample-2' 
tx = pd.read_csv(f'{data_dir}/400_transactions.csv')
tx.columns = tx.columns.str.strip()

hh = pd.read_csv(f'{data_dir}/400_households.csv')
hh.columns = hh.columns.str.strip()

hh = hh.drop(columns=['L', 'HH_SIZE', 'CHILDREN'])

# 2) Build RFM features
tx['PERIOD'] = tx['YEAR'] * 100 + tx['WEEK_NUM']
rfm = (
    tx
    .groupby('HSHD_NUM')
    .agg(
        recency_raw=('PERIOD', 'max'),
        frequency=('BASKET_NUM', 'nunique'),
        monetary=('SPEND', 'sum')
    )
    .reset_index()
)
max_period = tx['PERIOD'].max()
rfm['recency'] = max_period - rfm['recency_raw']
rfm.drop(columns='recency_raw', inplace=True)

# 3) Label churn = no purchase in last 12 weeks
rfm['churn'] = (rfm['recency'] > 12).astype(int)

# 4) Merge demographics & one-hot encode
data = rfm.merge(hh, on='HSHD_NUM', how='left')
cat_cols = ['AGE_RANGE', 'MARITAL', 'INCOME_RANGE', 'HOMEOWNER', 'HSHD_COMPOSITION']
data = pd.get_dummies(data, columns=cat_cols, drop_first=True)

# 5) Train/test split
X = data.drop(columns=['HSHD_NUM', 'churn'])
y = data['churn']
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42, stratify=y
)

# 6) Fit & evaluate logistic regression
model = LogisticRegression(solver='liblinear', max_iter=1000)
model.fit(X_train, y_train)

y_pred = model.predict(X_test)
y_proba = model.predict_proba(X_test)[:, 1]

print("Classification Report:\n", classification_report(y_test, y_pred))
auc = roc_auc_score(y_test, y_proba)
print(f"ROC AUC: {auc:.3f}")

# 7a) ROC curve
fpr, tpr, _ = roc_curve(y_test, y_proba)
plt.figure()
plt.plot(fpr, tpr, label=f'AUC = {auc:.3f}')
plt.plot([0,1], [0,1], linestyle='--')
plt.xlabel('False Positive Rate')
plt.ylabel('True Positive Rate')
plt.title('ROC Curve: Churn Prediction')
plt.legend(loc='lower right')
plt.tight_layout()
plt.show()
